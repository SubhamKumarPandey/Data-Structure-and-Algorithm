#include <bits/stdc++.h>
using namespace std;
void insert(stack<int> &s,int x)
{
 s.push(x);
}

void remove(stack<int> &s)
{
    
   s.pop();
}


void headOf_Stack(stack<int> &s)
{
    int x=s.top();
    cout<<x<<" "<<endl; 
}


bool find(stack<int> s, int val)
{
    bool exists=false;
    while(!s.empty())       
    {
        int top=s.top(); 
        s.pop();
        if(top==val)        
        exists=true;
    }
    
    if(exists==true){
        return true;
    }
    else{
        return false;
    }
}

	int testcases;

}
