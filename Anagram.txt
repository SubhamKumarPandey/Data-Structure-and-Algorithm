class Solution
{
    public:
    //Function is to check whether two strings are anagram of each other or not.
    const int CHAR=256;
    bool isAnagram(string a, string b){
        int count[CHAR]={0};
        if(a.length()!=b.length()){
            return false;
        }
        for(int j=0;j<a.length();j++){
            count[a[j]]++;
            count[b[j]]--;
        }
        for(int i=0;i<CHAR;i++){
            if(count[i]!=0){
                return false;
            }
        }
        return true;
        
        
        // Your code here
        
    }

};
